PHP and MySQL CRUD Library / Utility Scripts and Functions:
<?php /* This code offers an easier way to communicate with MySQL database without manually typing it. It comes with sanitation and pre-built console.log warnings and errors. CRUD: - Create - Read - Update - Delete */ $globalConn = null; /** * Establishes a connection to the MySQL database. * @param string $serverName The server name or IP address. * @param string $username The database username. * @param string $password The database password. * @param string $databaseName The database name. */ function MysqlConnectDB($serverName, $username, $password, $databaseName) { global $globalConn; $globalConn = new mysqli($serverName, $username, $password, $databaseName); if ($globalConn->connect_error) { echo "<script>console.log('Connection failed: " . addslashes($globalConn->connect_error) . "');</script>"; die("Connection failed: " . $globalConn->connect_error); } } /** * Sanitizes a string to prevent SQL injection. * @param string $string The string to sanitize. * @return string The sanitized string. */ function Sanitize($string) { global $globalConn; return $globalConn->real_escape_string($string); } /** * Creates a table in the database. * @param string $tableName The name of the table. * @param array $parameters An array of column definitions. */ function MysqlCreateTable($tableName, $parameters) { global $globalConn; $tableName = Sanitize($tableName); $sanitizedParameters = array_map('Sanitize', $parameters); $columns = implode(", ", $sanitizedParameters); $sql = "CREATE TABLE IF NOT EXISTS $tableName ($columns)"; if (!($stmt = $globalConn->prepare($sql))) { echo "<script>console.log('Error preparing table creation: " . addslashes($globalConn->error) . "');</script>"; } else { $stmt->execute(); $stmt->close(); } } /** * Inserts records into a table. * @param string $tableName The name of the table. * @param array $columns An array of column names. * @param array $values An array of arrays representing rows to insert. */ function MysqlInsert($tableName, $columns, $values) { global $globalConn; $tableName = Sanitize($tableName); $sanitizedColumns = array_map('Sanitize', $columns); $columnsString = implode(", ", $sanitizedColumns); $placeholders = rtrim(str_repeat('?, ', count($sanitizedColumns)), ', '); $sql = "INSERT INTO $tableName ($columnsString) VALUES ($placeholders)"; if (!($stmt = $globalConn->prepare($sql))) { echo "<script>console.log('Error preparing insert statement: " . addslashes($globalConn->error) . "');</script>"; } else { foreach ($values as $row) { $sanitizedRow = array_map('Sanitize', $row); $stmt->bind_param(str_repeat('s', count($sanitizedRow)), ...$sanitizedRow); $stmt->execute(); } $stmt->close(); } } /** * Selects data from a table. * @param string $tableName The name of the table. * @param string $column The column to retrieve. * @param string $condition Optional condition for filtering results. * @return string The retrieved data. */ function MysqlSelect($tableName, $column, $condition = "") { global $globalConn; $tableName = Sanitize($tableName); $column = Sanitize($column); $condition = Sanitize($condition); $whereClause = ""; if (!empty($condition)) { $whereClause = "WHERE $condition"; } $sql = "SELECT $column FROM $tableName $whereClause"; if (!($stmt = $globalConn->prepare($sql))) { echo "<script>console.log('Error preparing select statement: " . addslashes($globalConn->error) . "');</script>"; return ""; } else { $stmt->execute(); $result = $stmt->get_result(); $output = ""; while ($row = $result->fetch_assoc()) { $output .= reset($row); } $stmt->close(); return $output; } } /** * Updates records in a table. * @param string $tableName The name of the table. * @param string $column The column to update. * @param string $newValue The new value for the column. * @param string $condition The condition to identify records to update. */ function MysqlUpdate($tableName, $column, $newValue, $condition) { global $globalConn; $tableName = Sanitize($tableName); $column = Sanitize($column); $newValue = Sanitize($newValue); $condition = Sanitize($condition); $sql = "UPDATE $tableName SET $column = ? WHERE $condition"; if (!($stmt = $globalConn->prepare($sql))) { echo "<script>console.log('Error preparing update statement: " . addslashes($globalConn->error) . "');</script>"; } else { $stmt->bind_param('s', $newValue); $stmt->execute(); $stmt->close(); } } /** * Deletes records from a table. * @param string $tableName The name of the table. * @param string $condition The condition to identify records to delete. */ function MysqlDelete($tableName, $condition) { global $globalConn; $tableName = Sanitize($tableName); $condition = Sanitize($condition); $sql = "DELETE FROM $tableName WHERE $condition"; if (!($stmt = $globalConn->prepare($sql))) { echo "<script>console.log('Error preparing delete statement: " . addslashes($globalConn->error) . "');</script>"; } else { $stmt->execute(); $stmt->close(); } } /** * Executes a custom SQL query. * @param string $query The SQL query to execute. * @return bool True on success, false on failure. */ function MysqlCustomQuery($query) { global $globalConn; $sanitizedQuery = Sanitize($query); if (!($stmt = $globalConn->prepare($sanitizedQuery))) { echo "<script>console.log('Error preparing custom query: " . addslashes($globalConn->error) . "');</script>"; return false; } else { $stmt->execute(); $stmt->close(); return true; } } /* Syntax: MysqlConnectDB(); <Query> $globalConn->close(); ------------------------------------------------------------ Using this database configuration: $serverName = "localhost"; $username = "root"; $password = ""; $databaseName = "sample_database"; ------------------------------------------------------------ Example 1 (Create Table): MysqlConnectDB($serverName, $username, $password, $databaseName); MysqlCreateTable("users", [ "id INT PRIMARY KEY AUTO_INCREMENT", "first_name VARCHAR(200)", "middle_name VARCHAR(200)", "last_name VARCHAR(200)" ]); $globalConn->close(); ------------------------------------------------------------ Example 2 (Insert Data): MysqlConnectDB($serverName, $username, $password, $databaseName); MysqlInsert("users", ["first_name", "middle_name", "last_name"], [ ["john", "calysa", "doe"], ["jane", "calusin", "doe"], ["jake", "himenez", "zyrus"], ["sophia", "rain", "alkantara"], ["patrick", "monte", "reyes"], ["kenneth", "haque", "albino"] ]); $globalConn->close(); ------------------------------------------------------------ Example 3 (Select Data): MysqlConnectDB($serverName, $username, $password, $databaseName); $x = MysqlSelect("users", "first_name", "id = 3"); $globalConn->close(); echo $x; ------------------------------------------------------------ Example 4 (Update Data): MysqlConnectDB($serverName, $username, $password, $databaseName); MysqlUpdate("users", "first_name", "Jake", "id = 2"); $globalConn->close(); ------------------------------------------------------------ Example 5 (Delete Data): MysqlConnectDB($serverName, $username, $password, $databaseName); MysqlDelete("users", "id = 4"); $globalConn->close(); ------------------------------------------------------------ Example 6 (Custom Query): MysqlConnectDB($serverName, $username, $password, $databaseName); MysqlCustomQuery("ALTER TABLE users AUTO_INCREMENT = 1001"); $globalConn->close(); */ ?>
