Unity display text in user interface:
using UnityEngine; using UnityEngine.UI; public class DisplayTextScript : MonoBehaviour { // Public so that it can be defined in the Unity Editor's Inspector window public Text text1; // A private int to hold the number to display private int numberToDisplay = 42; void Start() { DisplayText(); // Calls the function to display the text text1.text = ""; // Empties the text in the user interface initially } // Method to display the text public void DisplayText() { text1.text = numberToDisplay.ToString(); // Converts int to string and sets the text } }
Unity playerprefs setting new highscore:
using UnityEngine; public class HighScoreManager : MonoBehaviour { public int score; private int highScore; void Start() { LoadHighScore(); // Example: Check if the current score is higher than the saved high score if (score > highScore) { highScore = score; SaveHighScore(); } } void LoadHighScore() { // Load the high score from PlayerPrefs, or set to 0 if it doesn't exist highScore = PlayerPrefs.GetInt("HighScore", 0); } void SaveHighScore() { // Save the high score to PlayerPrefs PlayerPrefs.SetInt("HighScore", highScore); PlayerPrefs.Save(); // Always save } }
