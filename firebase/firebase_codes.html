import { firestore } from './Firebase'; import { doc, setDoc, getDoc, onSnapshot, deleteDoc } from 'firebase/firestore'; import { useEffect, useState } from 'react'; // Define the type for the data you will be writing interface Data { [key: string]: any; } // Create / Update (Update if merge is true) const writeData = async (colId: string, docId: string, newData: Data, isMerge: boolean): Promise<void> => { const docRef = doc(firestore, colId, docId); try { await setDoc(docRef, newData, { merge: isMerge }); console.log("Data written successfully!"); } catch (e) { console.error("Data write failed:", e); } }; // Read (On Request) and return data const readDataOnRequest = async (colId: string, docId: string): Promise<Data | null> => { try { const docRef = doc(firestore, colId, docId); const docSnap = await getDoc(docRef); if (docSnap.exists()) { const data = docSnap.data(); console.log("Document data:", data); return data; } else { console.log("No such document!"); return null; } } catch (e) { console.error("Error getting document:", e); return null; } } // Read (Real Time) and return data const useRealTimeData = (colId: string, docId: string) => { const [data, setData] = useState<Data | null>(null); const [error, setError] = useState<string | null>(null); useEffect(() => { const docRef = doc(firestore, colId, docId); // Set up real-time listener const unsubscribe = onSnapshot(docRef, (docSnap) => { if (docSnap.exists()) { setData(docSnap.data() as Data); } else { setData(null); } }, (error) => { console.error("Error listening to document:", error); setError("Error listening to document."); }); // Clean up the listener on component unmount return () => unsubscribe(); }, [colId, docId]); return { data, error }; } // Delete a document const deleteData = async (colId: string, docId: string): Promise<void> => { const docRef = doc(firestore, colId, docId); try { await deleteDoc(docRef); console.log("Document deleted successfully!"); } catch (e) { console.error("Error deleting document:", e); } } export default function App() { // Use real-time data hook const { data: realTimeData, error } = useRealTimeData("SampleCollection", "SampleDocumentID"); useEffect(() => { const fetchData = async () => { // Example call to writeData await writeData( "SampleCollection", "SampleDocumentID", { email: "sample@gmail.com", }, false ); // Example call to readDataOnRequest and handle alert const data = await readDataOnRequest("SampleCollection", "SampleDocumentID"); if (data && data.email) { alert(`Email on request: ${data.email}`); } else { alert("Email not found."); } }; fetchData(); const removeData = async() => { await deleteData("SampleCollection", "SampleDocumentID"); } removeData(); }, []); useEffect(() => { if (realTimeData) { alert(`Real-time email: ${realTimeData.email}`); } else if (error) { alert(`Real-time data error: ${error}`); } }, [realTimeData, error]); return ( <div> {/* Your component JSX */} </div> ); }
