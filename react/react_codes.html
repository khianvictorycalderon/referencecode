3 Methods of Styling:
/* 3 methods to style in react: 1. External - Conflicts with class name but global 2. Modules - Requires additional setup but unique class names 3. Inline - Minimal styling component but less maintainable */ //-------------------------------------------------------- /* External: Simply import the css in main.tsx Example: import 'cssfile.css' Then use the className */ //---------------------- /* Modules: Using css file as a variable and generate unique className. Usually has a dedicated folder styling for a specific component. Example: */ import CustomStyle from './Sample.module.css' // The css file must have module name extension then css function ModuleStyling() { return ( <div className={CustomStyle.customStyleComponent}>Hi</div> ) } /* Sample.module.css Code: (For Demonstration purpose only) .customStyleComponent { background-color:red; padding:10px; } */ //---------------------- /* InLine: Specific styling that doesn't use a className All dash must be remove and capitalized the first letter after the dash: Examples: a.) border-radius = borderRadius b.) background-color = backgroundColor And values must be string and each style must be separated by comma Example of code: */ // The css file must have module name extension then css function InLineStyling() { const CustomSpecificStyle = { backgroundColor: "red", padding: "10px", } return ( <div style={CustomSpecificStyle}>Hi</div> ) } //-------------------------------------- export default function App() { return ( <> <InLineStyling /> <ModuleStyling /> </> ) }
